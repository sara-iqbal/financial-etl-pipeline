Fraud Rate by Transaction Category SQL Query
SELECT category,
  COUNT(*) AS total_txns,
  ROUND(AVG(is_fraud) * 100, 2) AS fraud_rate_pct,
  ROUND(AVG(amount), 2) AS avg_amount
FROM transactions
GROUP BY category
ORDER BY fraud_rate_pct DESC
